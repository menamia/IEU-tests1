Sub ValidateInputs()
    Dim ws As Worksheet
    Dim lastRow As Long
    Dim i As Long
    Dim valid As Boolean

    Set ws = ThisWorkbook.Sheets("Sheet1") ' Change to your sheet name
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row ' Find the last row with data

    ' Clear previous validation results
    ws.Range("E2:E" & lastRow).ClearContents ' Clear previous validation results

    ' Validate inputs
    valid = True
    For i = 2 To lastRow ' Assuming data starts from row 2
        If ws.Cells(i, 1).Value < 0 Or ws.Cells(i, 1).Value > 1000 Then ' Number_ID in column A
            ws.Cells(i, 5).Value = "Error" ' Mark error in column E
            valid = False
        End If
        If Not (ws.Cells(i, 2).Value = "Energy" Or ws.Cells(i, 2).Value = "Education" Or ws.Cells(i, 2).Value = "Water") Then ' Area in column B
            ws.Cells(i, 5).Value = "Error"
            valid = False
        End If
        If ws.Cells(i, 3).Value <= 0 Then ' Value 1 in column C
            ws.Cells(i, 5).Value = "Error"
            valid = False
        End If
        If ws.Cells(i, 4).Value <= 0 Then ' Value 2 in column D
            ws.Cells(i, 5).Value = "Error"
            valid = False
        End If
    Next i

    If valid Then
        MsgBox "Input validation completed successfully."
    Else
        MsgBox "Some inputs are invalid. Please check column E for details."
    End If
End Sub
